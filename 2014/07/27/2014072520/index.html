﻿这个方法可以将数组内元素相同的归位一类- (NSArray )sort:(NSArray )oldArr {
\tNSMutableArray newArr = [NSMutableArray array];
\tid tmpObj;
\ttmpObj = oldArr.firstObject;
\tNSMutableArray tmpArr = [NSMutableArray array];
\tfor (int i = 0; i < oldArr.count; i++) {
\t\tif ([tmpObj isEqual:oldArr[i]]) {
\t\t\t[tmpArr addObject:oldArr[i]];
\t\t}
\t\telse {
\t\t\t[newArr addObject:tmpArr];
\t\t\ttmpArr = [NSMutableArray array];
\t\t\t[tmpArr addObject:oldArr[i]];
\t\t\ttmpObj = oldArr[i];
\t\t}

}
