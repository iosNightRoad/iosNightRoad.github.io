instancesRespondToSelector与respondsToSelector注意点Only转载

﻿// main.m#import @interface Test : NSObject-(void) objectFun;+(void) classFun;@end@implementation Test-(void) objectFun{}+(void) classFun{}@endint main(){BOOL ret;ret=[Test instancesRespondToSelector:@selector(objectFun)]; //YESret=[Test instancesRespondToSelector:@selector(classFun)]; //NOret=[Test respondsToSelector:@selector(objectFun)]; //NOret=[Test respondsToSelector:@selector(classFun)]; //YES[Test classFun];//[Test objectFun];//error: Class method \’+objectFun\’ not foundTest* obj = [[Test alloc] init];ret=[obj respondsToSelector:@selector(objectFun)]; //YESret=[obj respondsToSelector:@selector(classFun)]; //NO//[obj classFun];//error: Instance method \’-classFun\’ not found[obj objectFun];return 0;}总结：1. respondsToSelector既可以检查类（是否响应指定类方法），也可以检查实例（是否响应指定实例方法）2. instancesRespondToSelector只能写在类名后面，但检测的是实例（是否响应指定实例方法）可以认为[Test instancesRespondToSelector:sel]等价于[obj respondsToSelector:sel]
