\n-(void) drawTextInRect:(CGRect)requestedRect
\n{
\n //去掉空行
\n NSString labelString = self.text;
\n NSString myString = [labelString stringByReplacingOccurrencesOfString:@\”\r\n\” withString:@\”\n\”];
\n //创建AttributeString
\n NSMutableAttributedString *string =[[NSMutableAttributedString alloc]initWithString:myString];
\n //设置字体及大小
\n CTFontRef helveticaBold = CTFontCreateWithName((CFStringRef)self.font.fontName,self.font.pointSize,NULL);
\n [string addAttribute:(id)kCTFontAttributeName value:(bridge id)helveticaBold range:NSMakeRange(0,[string length])];
\n //设置字间距
\n if(self.characterSpacing)
\n {
\n long number = self.characterSpacing;
\n CFNumberRef num = CFNumberCreate(kCFAllocatorDefault,kCFNumberSInt8Type,&number);
\n [string addAttribute:(id)kCTKernAttributeName value:(bridge id)num range:NSMakeRange(0,[string length])];
\n CFRelease(num);
\n }
\n //设置字体颜色
\n [string addAttribute:(id)kCTForegroundColorAttributeName value:(id)(self.textColor.CGColor) range:NSMakeRange(0,[string length])];
\n //创建文本对齐方式
\n CTTextAlignment alignment = kCTLeftTextAlignment;
\n if(self.textAlignment == UITextAlignmentCenter)
\n {
\n alignment = kCTCenterTextAlignment;
\n }
\n if(self.textAlignment == UITextAlignmentRight)
\n {
\n alignment = kCTRightTextAlignment;
\n }
\n CTParagraphStyleSetting alignmentStyle;
\n alignmentStyle.spec = kCTParagraphStyleSpecifierAlignment;
\n alignmentStyle.valueSize = sizeof(alignment);
\n alignmentStyle.value = &alignment;
\n //设置文本行间距
\n CGFloat lineSpace = self.linesSpacing;
\n CTParagraphStyleSetting lineSpaceStyle;
\n lineSpaceStyle.spec = kCTParagraphStyleSpecifierLineSpacingAdjustment;
\n lineSpaceStyle.valueSize = sizeof(lineSpace);
\n lineSpaceStyle.value =&lineSpace;
\n //设置文本段间距
\n CGFloat paragraphSpacing = 5.0;
\n CTParagraphStyleSetting paragraphSpaceStyle;
\n paragraphSpaceStyle.spec = kCTParagraphStyleSpecifierParagraphSpacing;
\n paragraphSpaceStyle.valueSize = sizeof(CGFloat);
\n paragraphSpaceStyle.value = ¶graphSpacing;

\n CTParagraphStyleRef style = CTParagraphStyleCreate(settings ,3);
\n //给文本添加设置
\n [string addAttribute:(id)kCTParagraphStyleAttributeName value:(__bridge id)style range:NSMakeRange(0 , [string length])];
\n //排版
\n CTFramesetterRef framesetter = CTFramesetterCreateWithAttributedString((CFAttributedStringRef)string);
\n CGMutablePathRef leftColumnPath = CGPathCreateMutable();
\n CGPathAddRect(leftColumnPath, NULL ,CGRectMake(0 , 0 ,self.bounds.size.width , self.bounds.size.height));
\n CTFrameRef leftFrame = CTFramesetterCreateFrame(framesetter,CFRangeMake(0, 0), leftColumnPath , NULL);
\n //翻转坐标系统（文本原来是倒的要翻转下）
\n CGContextRef context = UIGraphicsGetCurrentContext();
\n CGContextSetTextMatrix(context , CGAffineTransformIdentity);
\n CGContextTranslateCTM(context , 0 ,self.bounds.size.height);
\n CGContextScaleCTM(context, 1.0 ,-1.0);
\n //画出文本
\n CTFrameDraw(leftFrame,context);
\n //释放
\n CGPathRelease(leftColumnPath);
\n // CFReleale(framesetter);
\n // CFRelease(helveticaBold);
\n UIGraphicsPushContext(context);

\n}

tableView设置可以滚动的背景图Only转载

﻿设置滚动背景图时，只要保证背景图在tableview中处于最低层。
//iOS7 的一句就搞定了
[tableView insertSubview:tableBackGroundView atIndex:0];

//iOS6
[tableView insertSubview:tableBackGroundView atIndex:0];

  • (UITableViewCell )tableView:(UITableView )tableView cellForRowAtIndexPath:(NSIndexPath *)indexPath；
  • (UIView )tableView:(UITableView )tableView viewForFooterInSection:(NSInteger)section;
  • (UIView )tableView:(UITableView )tableView viewForHeaderInSection:(NSInteger)section;

通过颜色来生成一个纯色图片Only

﻿- (UIImage )buttonImageFromColor:(UIColor )color{
CGRect rect = CGRectMake(0, 0, self.frame.size.width, self.frame.size.height);
UIGraphicsBeginImageContext(rect.size);
CGContextRef context = UIGraphicsGetCurrentContext();
CGContextSetFillColorWithColor(context, [color CGColor]);
CGContextFillRect(context, rect);
UIImage *img = UIGraphicsGetImageFromCurrentImageContext();
UIGraphicsEndImageContext();
return img;
}
